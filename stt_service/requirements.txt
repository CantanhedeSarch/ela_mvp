# ==================================================
# Notas de instalação
# ==================================================
# 
# 1. Criar ambiente virtual:
#    python -m venv venv_stt
#    venv_stt\Scripts\activate
#
# 2. Instalar dependências:
#    pip install -r requirements.txt
#
# 3. Baixar modelo Vosk pt-BR:
#    - Acesse: https://alphacephei.com/vosk/models
#    - Recomendado: vosk-model-small-pt-0.3 (50MB) ou
#                   vosk-model-pt-fb-v0.1.1-20220516_2113 (1.6GB, maior precisão)
#    - Extrair em: models/vosk-model-small-pt-0.3/
#
# 4. Configurar variável de ambiente:
#    set VOSK_MODEL_PATH=models/vosk-model-small-pt-0.3
#
# 5. Executar serviço:
#    python -m stt_service.gateway_comunicacao
#    ou
#    uvicorn stt_service.gateway_comunicacao:app --host 0.0.0.0 --port 9100
#
# ==================================================
